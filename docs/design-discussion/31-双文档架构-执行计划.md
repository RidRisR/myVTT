# 双文档架构 实施计划

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将单 Y.Doc 拆分为 Public Doc + GM Secret Doc，在服务器层面阻断玩家获取 GM 机密数据。

**Architecture:** 服务器增加 Token 认证，WebSocket 连接路由到 `{roomId}:public` 或 `{roomId}:secret`。GM 客户端连接两个文档，通过合并层向下层组件提供统一数据视图。跨文档操作（现身/隐藏）由 GM 客户端执行。

**Tech Stack:** Yjs v13 / y-websocket v2 / React 19 / Vitest 4 / TypeScript 5.9

```javascript
wss.on('connection', (conn, req) => {
  const url = new URL(req.url, `http://${req.headers.host}`)
  const docPath = url.pathname.slice(1) // "abc12345:public" or "abc12345:secret"
  const token = url.searchParams.get('token')

  const [roomId, docType = 'public'] = docPath.split(':')

  // 验证房间存在
  const rooms = readRooms()
  const room = rooms.find((r) => r.id === roomId)
  if (!room) {
    console.warn(`Rejected connection to unknown room: ${roomId}`)
    conn.close(4404, 'Room not found')
    return
  }

  // 验证文档类型
  if (docType !== 'public' && docType !== 'secret') {
    conn.close(4400, 'Invalid document type')
    return
  }

  // 过渡期：无 token 时允许连接 public（向后兼容）
  if (token) {
    const isGM = token === room.gmToken
    const isPlayer = token === room.playerToken
    if (!isGM && !isPlayer) {
      conn.close(4401, 'Invalid token')
      return
    }
    if (docType === 'secret' && !isGM) {
      conn.close(4403, 'Forbidden')
      return
    }
  } else if (docType === 'secret') {
    // secret 始终要求 token
    conn.close(4401, 'Token required')
    return
  }

  setupWSConnection(conn, req, { docName: docPath })
})
```

```javascript
bindState: async (docName, ydoc) => {
  const persistedYdoc = await ldb.getYDoc(docName)
  let newUpdates = Y.encodeStateAsUpdate(persistedYdoc)

  // 迁移：{roomId}:public 首次加载时从裸 roomId 继承数据
  if (docName.endsWith(':public') && persistedYdoc.store.clients.size === 0) {
    const bareRoomId = docName.replace(':public', '')
    try {
      const legacyDoc = await ldb.getYDoc(bareRoomId)
      if (legacyDoc.store.clients.size > 0) {
        console.log(`[migration] ${docName}: inheriting from legacy doc ${bareRoomId}`)
        newUpdates = Y.encodeStateAsUpdate(legacyDoc)
      }
    } catch (_) { /* no legacy doc */ }
  }

  Y.applyUpdate(ydoc, newUpdates)

  // roster → entities 迁移（仅对非 secret 文档）
  if (!docName.endsWith(':secret')) {
    const roster = ydoc.getMap('roster')
    const entities = ydoc.getMap('entities')
    if (roster.size > 0 && entities.size === 0) {
      ydoc.transact(() => {
        roster.forEach((val, key) => entities.set(key, val))
      })
      console.log(`[migration] ${docName}: migrated ${roster.size} entries from roster → entities`)
    }
  }

  ydoc.on('update', (update) => { ldb.storeUpdate(docName, update) })
},

// --- Secret Doc ---

export interface SecretWorldMaps {
  /** 完全隐藏的实体（格式同 public entities） */
  secretEntities: Y.Map<Y.Map<unknown>>
  /** 覆盖层：按 entityId 索引的秘密数据 */
  entitySecrets: Y.Map<Y.Map<unknown>>
  /** GM 自由笔记 */
  gmNotes: Y.Map<unknown>
}

export function createSecretWorldMaps(secretDoc: Y.Doc): SecretWorldMaps {
  return {
    secretEntities: secretDoc.getMap('secret_entities') as Y.Map<Y.Map<unknown>>,
    entitySecrets: secretDoc.getMap('entity_secrets') as Y.Map<Y.Map<unknown>>,
    gmNotes: secretDoc.getMap('gm_notes'),
  }
}

export function useSecretWorld(secretDoc: Y.Doc | null): SecretWorldMaps | null {
  return useMemo(() => (secretDoc ? createSecretWorldMaps(secretDoc) : null), [secretDoc])
}

// src/gm/crossDocOps.ts
import * as Y from 'yjs'
import { readYMapEntity, setYMapFields } from '../entities/useEntities'
import { defaultNPCPermissions } from '../shared/permissions'
import type { WorldMaps, SecretWorldMaps } from '../yjs/useWorld'

type PublicWorld = { yDoc: Y.Doc } & WorldMaps
type SecretWorld = { yDoc: Y.Doc } & SecretWorldMaps

/**
 * 现身：将隐藏实体从 secret doc 搬到 public doc。
 * 幂等 — 如果 public 已有该实体则跳过写入。
 */
export function revealEntity(entityId: string, secret: SecretWorld, pub: PublicWorld) {
  // 如果 public 已有，仅清理 secret
  if (pub.entities.get(entityId) instanceof Y.Map) {
    secret.secretEntities.delete(entityId)
    return
  }

  const secretYMap = secret.secretEntities.get(entityId)
  if (!(secretYMap instanceof Y.Map)) return

  const entity = readYMapEntity(secretYMap)
  // 现身后设为默认可见权限
  entity.permissions = defaultNPCPermissions()

  pub.yDoc.transact(() => {
    const yMap = new Y.Map<unknown>()
    pub.entities.set(entity.id, yMap)
    setYMapFields(yMap, entity)
  })

  secret.secretEntities.delete(entityId)
}

/**
 * 隐藏：将实体从 public doc 搬到 secret doc。
 * 同时从所有场景的 entityIds 中移除，删除关联 token。
 * 幂等 — 如果 secret 已有该实体则跳过。
 */
export function hideEntity(entityId: string, pub: PublicWorld, secret: SecretWorld) {
  // 如果 secret 已有，仅清理 public
  if (secret.secretEntities.get(entityId) instanceof Y.Map) {
    cleanupEntityFromPublic(entityId, pub)
    return
  }

  const publicYMap = pub.entities.get(entityId)
  if (!(publicYMap instanceof Y.Map)) return

  const entity = readYMapEntity(publicYMap)

  // 写入 secret
  secret.yDoc.transact(() => {
    const yMap = new Y.Map<unknown>()
    secret.secretEntities.set(entity.id, yMap)
    setYMapFields(yMap, entity)
  })

  // 从 public 清理
  cleanupEntityFromPublic(entityId, pub)
}

/** 从所有场景 entityIds 移除、删除关联 token、删除实体 */
function cleanupEntityFromPublic(entityId: string, pub: PublicWorld) {
  pub.yDoc.transact(() => {
    pub.scenes.forEach((sceneMap) => {
      if (!(sceneMap instanceof Y.Map)) return
      const entityIds = sceneMap.get('entityIds')
      if (entityIds instanceof Y.Map) entityIds.delete(entityId)
      const tokens = sceneMap.get('tokens')
      if (tokens instanceof Y.Map) {
        tokens.forEach((token: unknown, tokenId: string) => {
          if (
            token &&
            typeof token === 'object' &&
            (token as Record<string, unknown>).entityId === entityId
          ) {
            tokens.delete(tokenId)
          }
        })
      }
    })
    pub.entities.delete(entityId)
  })
}
```

// src/gm/useEntitySecrets.ts
import { useCallback, useEffect, useState } from 'react'
import * as Y from 'yjs'
import type { SecretWorldMaps } from '../yjs/useWorld'

export interface EntitySecretData {
  hpCurrent?: number
  hpMax?: number
  secretNotes?: string
}

/**
 * GM-only hook：读写 entity_secrets 覆盖层。
 * 返回 null 给非 GM 客户端。
 */
export function useEntitySecrets(secretWorld: SecretWorldMaps | null) {
  const [version, setVersion] = useState(0)

  useEffect(() => {
    if (!secretWorld) return
    const onChange = () => setVersion((v) => v + 1)
    secretWorld.entitySecrets.observeDeep(onChange)
    return () => secretWorld.entitySecrets.unobserveDeep(onChange)
  }, [secretWorld])

  const getSecretData = useCallback(
    (entityId: string): EntitySecretData | null => {
      if (!secretWorld) return null
      const yMap = secretWorld.entitySecrets.get(entityId)
      if (!(yMap instanceof Y.Map)) return null
      return {
        hpCurrent: yMap.get('hp_current') as number | undefined,
        hpMax: yMap.get('hp_max') as number | undefined,
        secretNotes: yMap.get('secret_notes') as string | undefined,
      }
    },
    // eslint-disable-next-line react-hooks/exhaustive-deps
    [secretWorld, version],
  )

  const setSecretData = useCallback(
    (entityId: string, data: Partial<EntitySecretData>) => {
      if (!secretWorld) return
      let yMap = secretWorld.entitySecrets.get(entityId)
      if (!(yMap instanceof Y.Map)) {
        yMap = new Y.Map<unknown>()
        secretWorld.entitySecrets.set(entityId, yMap)
      }
      if (data.hpCurrent !== undefined) yMap.set('hp_current', data.hpCurrent)
      if (data.hpMax !== undefined) yMap.set('hp_max', data.hpMax)
      if (data.secretNotes !== undefined) yMap.set('secret_notes', data.secretNotes)
    },
    [secretWorld],
  )

```typescript
// src/gm/useGmMergedView.ts
import { useMemo, useEffect, useState, useCallback } from 'react'
import * as Y from 'yjs'
import type { Entity } from '../shared/entityTypes'
import type { SecretWorldMaps } from '../yjs/useWorld'
import { readYMapEntity } from '../entities/useEntities'

export interface MergedEntity extends Entity {
  _hidden?: boolean
}

/**
 * GM 合并层：将 public entities 和 secret entities 合并为统一列表。
 * 隐藏实体带 _hidden = true 标记。
 * 非 GM 客户端（secretWorld 为 null）直接返回 publicEntities。
 */
export function useGmMergedView(
  publicEntities: Entity[],
  secretWorld: SecretWorldMaps | null,
): MergedEntity[] {
  const [secretEntities, setSecretEntities] = useState<Entity[]>([])

  // 监听 secret_entities 变化
  useEffect(() => {
    if (!secretWorld) return
    const rebuild = () => {
      const result: Entity[] = []
      secretWorld.secretEntities.forEach((yMap) => {
        if (yMap instanceof Y.Map) {
          result.push(readYMapEntity(yMap))
        }
      })
      setSecretEntities(result)
    }
    rebuild()
    secretWorld.secretEntities.observeDeep(rebuild)
    return () => secretWorld.secretEntities.unobserveDeep(rebuild)
  }, [secretWorld])

  return useMemo(() => {
    if (!secretWorld) return publicEntities
    const hidden: MergedEntity[] = secretEntities.map((e) => ({ ...e, _hidden: true }))
    return [...publicEntities, ...hidden]
  }, [publicEntities, secretEntities, secretWorld])
}
```

```typescript
// src/yjs/useDualDocConnection.ts
import { useEffect, useState } from 'react'
import * as Y from 'yjs'
import { WebsocketProvider } from 'y-websocket'
import type { Awareness } from 'y-protocols/awareness'
import { WEBSOCKET_URL } from '../shared/config'

export function useDualDocConnection(roomId: string, token: string | null) {
  const [publicDoc] = useState(() => new Y.Doc())
  const [secretDoc] = useState(() => new Y.Doc())
  const [isLoading, setIsLoading] = useState(true)
  const [awareness, setAwareness] = useState<Awareness | null>(null)

  const isGM = token ? token.startsWith('gm_') : false

  useEffect(() => {
    const params: Record<string, string> = {}
    if (token) params.token = token

    const publicProvider = new WebsocketProvider(WEBSOCKET_URL, `${roomId}:public`, publicDoc, {
      params,
    })
    setAwareness(publicProvider.awareness)

    let secretProvider: WebsocketProvider | null = null
    let publicSynced = false
    let secretSynced = !isGM // 非 GM 不需要等 secret

    const checkReady = () => {
      if (publicSynced && secretSynced) setIsLoading(false)
    }

    publicProvider.on('sync', (synced: boolean) => {
      if (synced) {
        publicSynced = true
        checkReady()
      }
    })

    if (isGM) {
      secretProvider = new WebsocketProvider(WEBSOCKET_URL, `${roomId}:secret`, secretDoc, {
        params,
      })
      secretProvider.on('sync', (synced: boolean) => {
        if (synced) {
          secretSynced = true
          checkReady()
        }
      })
    }

    return () => {
      publicProvider.destroy()
      secretProvider?.destroy()
      setAwareness(null)
    }
  }, [publicDoc, secretDoc, roomId, token, isGM])

  return {
    publicDoc,
    secretDoc: isGM ? secretDoc : null,
    isLoading,
    awareness,
    isGM,
  }
}
```
